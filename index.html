üìä Plotly Basics

Plotly offers two main approaches: Plotly Express (high-level) and Graph Objects (low-level).

Plotly Express - Quick Start

import plotly.express as px
import pandas as pd
# Sample data
df = pd.DataFrame({ 'x': [1, 2, 3, 4, 5], 'y': [10, 11, 12, 13, 14], 'category': ['A', 'B', 'A', 'B', 'A'] })
# Create a simple line chart
fig = px.line(df, x='x', y='y', color='category', title='My First Plotly Chart')
fig.show()

Graph Objects - More Control

import plotly.graph_objects as go
# Create figure with custom styling
fig = go.Figure()
fig.add_trace(go.Scatter( x=[1, 2, 3, 4, 5], y=[10, 11, 12, 13, 14], mode='lines+markers', name='Data Series', line=dict(color='blue', width=3) ))
fig.update_layout( title='Custom Styled Chart', xaxis_title='X Axis', yaxis_title='Y Axis' )
fig.show()

üìà Chart Types

1. Bar Charts

# Simple bar chart
df_sales = pd.DataFrame({ 'product': ['A', 'B', 'C', 'D'], 'sales': [100, 150, 120, 180] })
fig = px.bar(df_sales, x='product', y='sales', title='Product Sales', color='sales', color_continuous_scale='viridis')
fig.show()

2. Scatter Plots

# Scatter plot with size and color
df_bubble = pd.DataFrame({ 'x': [1, 2, 3, 4, 5],
'y': [10, 25, 30, 45, 50],
'size': [20, 40, 60, 80, 100],
'color': ['red', 'blue', 'green', 'orange', 'purple'] })
fig = px.scatter(df_bubble, x='x', y='y', size='size', color='color', title='Bubble Chart Example')
fig.show()

3. Heatmaps

# Correlation heatmap
import numpy as np
# Generate sample correlation matrix
corr_matrix = np.random.rand(5, 5)
labels = ['Var1', 'Var2', 'Var3', 'Var4', 'Var5']
fig = px.imshow(corr_matrix, labels=dict(x="Variables", y="Variables"), x=labels, y=labels, title='Correlation Heatmap')
fig.show()

üåê Dash Introduction

Dash allows you to build web applications using pure Python. No HTML, CSS, or JavaScript required!

Basic Dash App Structure

import dash from dash import dcc, html import plotly.express as px # Initialize the Dash app app = dash.Dash(__name__) # Sample data for chart df = px.data.iris() # Create a figure fig = px.scatter(df, x="sepal_width", y="sepal_length", color="species", title="Iris Dataset") # Define the app layout app.layout = html.Div([ html.H1("My First Dash App", style={'textAlign': 'center'}), dcc.Graph( id='iris-scatter', figure=fig ) ]) # Run the app if __name__ == '__main__': app.run_server(debug=True)

üîÑ Callbacks - Adding Interactivity

Callbacks make your dashboard interactive by connecting components together.

Basic Callback Example

from dash import Input, Output, callback # Callback decorator @app.callback( Output('output-div', 'children'), Input('input-box', 'value') ) def update_output(input_value): return f'You entered: {input_value}' # Layout with input and output app.layout = html.Div([ dcc.Input(id='input-box', type='text', value='Initial value'), html.Div(id='output-div') ])

Interactive Chart Callback

@app.callback( Output('graph', 'figure'), [Input('dropdown', 'value'), Input('slider', 'value')] ) def update_graph(dropdown_value, slider_value): # Filter data based on inputs filtered_df = df[df['category'] == dropdown_value] # Create updated figure fig = px.scatter(filtered_df, x='x', y='y', size=slider_value, title=f'Chart for {dropdown_value}') return fig

üéØ Complete Dashboard Example

Let's build a comprehensive sales dashboard with multiple interactive components.

import dash
from dash import dcc, html, Input, Output
import plotly.express as px
import pandas as pd
import numpy as np
# Generate sample sales data
np.random.seed(42)
dates = pd.date_range('2023-01-01', periods=365, freq='D')
df_sales = pd.DataFrame({
'date': dates,
'sales': np.random.randint(1000, 5000, 365),
'product': np.random.choice(['A', 'B', 'C', 'D'], 365),
'region': np.random.choice(['North', 'South', 'East', 'West'], 365)
})
# Initialize the app
app = dash.Dash(__name__)
# Define the layout
app.layout = html.Div([
html.H1("Sales Dashboard",
style={'textAlign': 'center', 'marginBottom': 30}),
# Controls row
html.Div([
html.Div([
html.Label("Select Product:"),
dcc.Dropdown(
id='product-dropdown',
options=[{'label': prod, 'value': prod}
for prod in df_sales['product'].unique()],
value=df_sales['product'].unique()[0]
)
], style={'width': '48%', 'display': 'inline-block'}),
html.Div([
html.Label("Select Region:"),
dcc.Dropdown(
id='region-dropdown',
options=[{'label': region, 'value': region}
for region in df_sales['region'].unique()],
value=df_sales['region'].unique()[0]
)
], style={'width': '48%', 'float': 'right', 'display': 'inline-block'})
]),
# Charts row
html.Div([
dcc.Graph(id='time-series-chart'),
dcc.Graph(id='bar-chart')
], style={'marginTop': 30})
])
# Callback for time series chart
@app.callback(
Output('time-series-chart', 'figure'),
[Input('product-dropdown', 'value'),
Input('region-dropdown', 'value')]
)
def update_time_series(selected_product, selected_region):
filtered_df = df_sales[
(df_sales['product'] == selected_product) &
(df_sales['region'] == selected_region)
]
fig = px.line(filtered_df, x='date', y='sales',
title=f'Sales Trend - {selected_product} in {selected_region}')
return fig
# Callback for bar chart
@app.callback(
Output('bar-chart', 'figure'),
[Input('product-dropdown', 'value')]
)
def update_bar_chart(selected_product):
filtered_df = df_sales[df_sales['product'] == selected_product]
region_sales = filtered_df.groupby('region')['sales'].sum().reset_index()

fig = px.bar(region_sales, x='region', y='sales',
title=f'Total Sales by Region - Product {selected_product}')
return fig
# Run the app
if __name__ == '__main__':
app.run_server(debug=True)
üöÄ Quick Start: Save this code as 'app.py' and run it with 'python app.py'. Open http://127.0.0.1:8050/ in your browser!

üöÄ Advanced Topics

1. Multi-Page Apps

from dash import dcc, html, Input, Output import dash app = dash.Dash(__name__) # Multi-page layout app.layout = html.Div([ dcc.Location(id='url', refresh=False), html.Div(id='page-content') ]) # Page layouts page_1_layout = html.Div([ html.H1('Page 1'), dcc.Link('Go to Page 2', href='/page-2') ]) page_2_layout = html.Div([ html.H1('Page 2'), dcc.Link('Go to Page 1', href='/page-1') ]) # Callback for page routing @app.callback(Output('page-content', 'children'), Input('url', 'pathname')) def display_page(pathname): if pathname == '/page-1': return page_1_layout elif pathname == '/page-2': return page_2_layout else: return page_1_layout

2. Real-time Updates

from dash import dcc import datetime # Add interval component for real-time updates app.layout = html.Div([ dcc.Graph(id='live-graph'), dcc.Interval( id='interval-component', interval=1000, # Update every second n_intervals=0 ) ]) @app.callback( Output('live-graph', 'figure'), Input('interval-component', 'n_intervals') ) def update_live_graph(n): # Generate random data x = list(range(n-10, n)) y = [np.random.randint(0, 100) for _ in range(10)] fig = px.line(x=x, y=y, title='Live Data Stream') return fig

3. Custom CSS Styling

# Custom CSS styles external_stylesheets = ['https://codepen.io/chriddyp/pen/bWLwgP.css'] app = dash.Dash(__name__, external_stylesheets=external_stylesheets) # Styled layout app.layout = html.Div([ html.H1("Styled Dashboard", style={ 'textAlign': 'center', 'color': '#2c3e50', 'backgroundColor': '#ecf0f1', 'padding': '20px', 'borderRadius': '10px' }), html.Div([ dcc.Graph(id='styled-graph') ], className='six columns'), ], className='container')

4. Data Persistence

# Store component for data persistence from dash import dcc app.layout = html.Div([ dcc.Store(id='session-data'), dcc.Input(id='input', type='text'), html.Div(id='output') ]) @app.callback( Output('session-data', 'data'), Input('input', 'value') ) def store_data(value): return {'stored_value': value} @app.callback( Output('output', 'children'), Input('session-data', 'data') ) def display_stored_data(data): if data: return f"Stored: {data.get('stored_value', '')}" return "No data stored"
